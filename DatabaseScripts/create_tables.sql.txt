CREATE DATABASE ticket_system;
USE ticket_system;

CREATE TABLE Users (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(100),
    Email VARCHAR(100),
    Password VARCHAR(100),
    Role VARCHAR(50)
);

CREATE TABLE Tickets (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    TicketNumber VARCHAR(50),
    Title VARCHAR(255),
    Description TEXT,
    Status VARCHAR(50),
    Priority VARCHAR(50),
    CreatedBy INT,
    AssignedTo INT,
    CreatedAt DATETIME
);

CREATE TABLE Comments (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    TicketId INT,
    Message TEXT,
    CreatedBy INT,
    CreatedAt DATETIME
);
